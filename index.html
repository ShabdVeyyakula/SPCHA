public class SuperCleanupKarel extends SuperKarel
{
    

    
    public void run()
    {
        
        int height = 1;
        int width = 1;
        int direction = 0;
        
        //Calculates Width
        while (frontIsClear())
        {
            move();
            width = width + 1;
        }
        
        turnAround();
        
        while (frontIsClear())
        {
            move();
        }
        
        turnAround();
        
        turnLeft();
        
        //Calculates Height
        
        while(frontIsClear())
        {
            move();
            height = height + 1;
        }
        
        turnAround();
        
        while (frontIsClear())
        {
            move();
        }
        
        turnAround();
        
        turnRight();
        
        for(int i = 0; i < height - 1; i ++)
        {
            for(int x = 0; x < width ; x++)
            {
                if(ballsPresent())
                {
                    takeBall();
                }
                
                if(frontIsClear())
                {
                    move();
                }
                
            }
            
            if(direction == 1)
            {
                turnRight();
                move();
                turnRight();  
                direction = 0;
            }
            
            else if(direction == 0)
            {
                turnLeft();
                move();
                turnLeft();  
                direction = 1;
            }
            
            
            
        }
        
        for(int x = 0; x < width ; x++)
        {
            if(ballsPresent())
            {
                takeBall();
            }
                
            if(frontIsClear())
            {
                move();
            }
        }
            
    }
    
    public void doNothing()
    {
        move();
    }

}
